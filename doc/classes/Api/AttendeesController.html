def create
  @attendee = Attendee.new(attendee_params)

  if @attendee.save
    render :status => 200,
      :json => { :success => true,
                 :info => "Attendee Created",
                 :data => {} }
    else
      render :status => :unprocessable_entity,
        :json => { :success => false,
                   :info => resource.errors,
                   :data => {} }
        end
  end
